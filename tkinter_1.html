Пример

import tkinter as tk

def on_click():
    label.config(text=entry.get())

root = tk.Tk()
label = tk.Label(root, text="Привет")
entry = tk.Entry(root)
button = tk.Button(root, text="Показать", command=on_click)

label.pack()
entry.pack()
button.pack()

root.mainloop()
